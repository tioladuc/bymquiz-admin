SELECT 
GROUP_CONCAT(DISTINCT id ORDER BY id ASC SEPARATOR '; ') AS ids,
title, question, explication, description, bible_references, count(*) as total
FROM `data_mcq_tmp` 
Group by title, question, explication, description, bible_references 
having count(*)>1
*************************SUPPRESSION DOUBLONS******************
DELETE t1 FROM data_mcq_tmp t1 JOIN data_mcq_tmp t2 ON t1.title = t2.title AND t1.question = t2.question AND t1.explication = t2.explication AND t1.description = t2.description AND t1.bible_references = t2.bible_references AND t1.id > t2.id

drop table data_mcq_validate;
drop table data_mcq_tmp;
drop table data_mcq;
